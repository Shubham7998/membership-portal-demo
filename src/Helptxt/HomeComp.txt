export default function GenericListSubscription({
    data,
    handleDelete,
    handleEdit,
    subscriberInfo,
    dataHeader,
    isSearchMode,
    tableName,
    handleSorting
}: GenericListProps) {
    const getSubscriberName = (id: number): string => {
        const subscriber = subscriberInfo.find(subscriber => subscriber.id === id);
        if (subscriber) {
            return `${subscriber.firstName} ${subscriber.lastName}`;
        }
        return 'Subscriber Not Found';
    };

    return (
        <>
            <TableContainer component={Paper}>
                <Table sx={{ minWidth: 700 }} aria-label="customized table">
                    {/* Table headers */}
                    <TableHead>
                        <TableRow>
                            {/* Render table headers */}
                        </TableRow>
                    </TableHead>
                    {/* Table body */}
                    <TableBody>
                        {data?.map((item: any, index) => (
                            <StyledTableRow key={index}>
                                <StyledTableCell align="left">{/* Render Sr No */}</StyledTableCell>
                                {Object.entries(item).map(([key, value]: any, idx) => {
                                    if (key !== "id" && key !== "discountId" && key !== "taxId" && key !== "productId" && key !== "genderId") {
                                        if (key === 'subscriberId') {
                                            return (
                                                <StyledTableCell key={idx} align="center">
                                                    {getSubscriberName(value)}
                                                </StyledTableCell>
                                            );
                                        }
                                        return (
                                            <StyledTableCell key={idx} align="center">
                                                {/* Render other data */}
                                            </StyledTableCell>
                                        );
                                    }
                                })}
                                {/* Render edit and delete buttons */}
                            </StyledTableRow>
                        ))}
                    </TableBody>
                </Table>
            </TableContainer>
        </>
    );
}
