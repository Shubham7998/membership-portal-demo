import React from 'react';
import { Grid, FormControl, InputLabel, Select, MenuItem } from '@mui/material';

interface Product {
  id: number;
  productName: string;
}

interface Props {
  productInfo: Product[];
  productInfoSearch: { productName: string };
  handleSelectChange: (event: React.ChangeEvent<{ name?: string; value: unknown }>) => void;
}

const YourComponent: React.FC<Props> = ({ productInfo, productInfoSearch, handleSelectChange }) => {
  return (
    <Grid item xs={12} sx={{ marginRight: 5 }}>
      <FormControl fullWidth>
        <InputLabel id="productName-label">Product Name</InputLabel>
        <Select
          labelId="productName-label"
          id="productName"
          value={productInfoSearch.productName.toString()} 
          label="Product Name"
          name='productName'
          required
          onChange={handleSelectChange}
        >
          <MenuItem value={""} >---Select Product Name---</MenuItem>
          {productInfo?.map((product) => (
            <MenuItem key={product.id} value={product.id}>{product.productName}</MenuItem>
          ))}
        </Select>
      </FormControl>
    </Grid>
  );
}

export default YourComponent;
